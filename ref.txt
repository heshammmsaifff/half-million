// ุงูููููุงุช ุงููุณุงุนุฏุฉ ุงูุฎุงุฑุฌูุฉ
function InputGroup({
  label,
  value,
  onChange,
  type = "text",
  placeholder = "",
}) {
  return (
    <div className="text-right">
      <label className="text-sm font-bold text-gray-700 mb-2 block">
        {label}
      </label>
      <input
        type={type}
        value={value}
        placeholder={placeholder}
        onChange={(e) => onChange(e.target.value)}
        className="w-full bg-white p-4 rounded-2xl border-2 border-gray-200 outline-none focus:border-blue-500 focus:ring-4 ring-blue-50 transition-all font-bold"
      />
    </div>
  );
}

function TextAreaGroup({ label, value, onChange, height }) {
  return (
    <div className="text-right">
      <label className="text-sm font-bold text-gray-700 mb-2 block">
        {label}
      </label>
      <textarea
        value={value}
        onChange={(e) => onChange(e.target.value)}
        className={`w-full bg-white p-4 rounded-2xl border-2 border-gray-200 outline-none focus:border-blue-500 focus:ring-4 ring-blue-50 transition-all font-medium ${height}`}
      />
    </div>
  );
}

function DynamicList({
  label,
  items,
  onAdd,
  onUpdate,
  onRemove,
  setter,
  color,
}) {
  const colors = {
    purple: "text-purple-600 bg-purple-50",
    green: "text-green-600 bg-green-50",
    blue: "text-blue-600 bg-blue-50",
    red: "text-red-600 bg-red-50",
  };
  return (
    <div className="space-y-4">
      <div className="flex justify-between items-center">
        <label className="font-bold text-gray-800">{label}</label>
        <button
          type="button"
          onClick={onAdd}
          className={`p-1.5 rounded-xl ${colors[color]}`}
        >
          <Plus size={20} />
        </button>
      </div>
      <div className="space-y-2">
        {items.map((it, idx) => (
          <div key={idx} className="flex gap-2">
            <input
              value={it}
              onChange={(e) => onUpdate(idx, e.target.value, setter)}
              className="flex-1 p-3 bg-white border-2 border-gray-100 rounded-xl text-sm font-bold"
            />
            <button
              type="button"
              onClick={() => onRemove(idx, setter)}
              className="text-gray-300 hover:text-red-500"
            >
              <Trash2 size={18} />
            </button>
          </div>
        ))}
      </div>
    </div>
  );
}

const addItem = (setter) => setter((prev) => [...prev, ""]);
const updateItem = (index, value, setter) => {
  setter((prev) => {
    const newArr = [...prev];
    newArr[index] = value;
    return newArr;
  });
};
const removeItem = (index, setter) =>
  setter((prev) => prev.filter((_, i) => i !== index));
